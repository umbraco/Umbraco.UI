# Icon Registry Essential

Essential Icon Registry of Umbraco UI Library. This component delivers essential Umbraco UI icons for any child of this component.

<br />

<uui-icon-registry-essential>
  <IconGallery>
    {Object.entries(icons).map(([name]) => {
      const processedName = name.replace('icon', '').toLowerCase();

      return (
        <IconItem name={processedName} key={processedName}>
          <uui-icon name={processedName}></uui-icon>
        </IconItem>
      );
    })}

  </IconGallery>
</uui-icon-registry-essential>
